Laravel Notes

// create new laravel project
composer create-project laravel/laravel example-app

// change directory
cd example-app

// start hosting
php artisan serve

// for installing bootstrap
composer require laravel/ui

// install bootstrap
php artisan ui bootstrap

// bootstrap
npm i bootstrap@5.3.2

// check node installation
# for node
node -v

# for npm
npm -v

# for php
php -v 

# for composer
composer -v

// run bootstrap
npm install
npm run dev

// storage link for pictures 
php artisan storage:link

// Bootstrap link
<!-- Scripts -->
    @vite(['resources/sass/app.scss', 'resources/js/app.js'])

// create views
php artisan make:view {name}

// create model, composer and migration/database
php artisan make:model Product -cm

// upload database to localhost
php artisan migrate
